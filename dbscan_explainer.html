▸ Python Implementation

Two ways to run DBSCAN: from scratch (educational) and via scikit-learn (production).

dbscan_from_scratch.py 
import math

def region_query(points, p, eps):
    """Return indices of all points within eps of points[p]."""
    return [i for i, q in enumerate(points)
            if math.dist(points[p], q) <= eps]

def dbscan(points, eps, min_pts):
    labels  = [-1] * len(points)   # -1 = noise
    visited = [False] * len(points)
    cluster_id = 0

    for i in range(len(points)):
        if visited[i]:
            continue
        visited[i] = True
        neighbors = region_query(points, i, eps)

        if len(neighbors) < min_pts:
            labels[i] = -1          # mark noise for now
            continue

        # i is a core point — start new cluster
        labels[i] = cluster_id
        queue = [n for n in neighbors if n != i]

        while queue:
            q = queue.pop(0)
            if not visited[q]:
                visited[q] = True
                q_neighbors = region_query(points, q, eps)
                if len(q_neighbors) >= min_pts:  # q is also core
                    queue += [n for n in q_neighbors
                              if n not in queue]
            if labels[q] == -1:            # unassigned → add to cluster
                labels[q] = cluster_id

        cluster_id += 1

    return labels

sklearn_dbscan.py 
import numpy as np
from sklearn.cluster import DBSCAN

POINTS = np.array([
    [220,230],[250,210],[250,250],[220,270],[190,250],
    [190,210],[290,190],[290,270],[220,310],[150,270],
    [150,190],[220,180],[560,230],[590,210],[590,250],
    [560,270],[530,250],[530,210],[630,190],[630,270],
    [560,310],[490,270],[490,190],[560,180],[390,100],
    [390,360],[80,150],[80,310],[700,150],[700,310],
])

db = DBSCAN(eps=45, min_samples=4).fit(POINTS)
labels = db.labels_   # -1 = noise

n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
n_noise    = (labels == -1).sum()

print(f"Clusters: {n_clusters}")   # 2
print(f"Noise:    {n_noise}")       # 6
print(f"Labels:   {labels}")

# which points are core points
core_mask = np.zeros(len(POINTS), dtype=bool)
core_mask[db.core_sample_indices_] = True
