dbscan_from_scratch.py 
#!/usr/bin/env python3
"""Raw DBSCAN — ε = 45 (pixel distance), MinPts = 4."""

import math
from collections import deque


def euclidean(a, b):
    return math.sqrt((a[0] - b[0]) ** 2 + (a[1] - b[1]) ** 2)


def region_query(points, p_idx, eps):
    """Return indices of all points within ε of points[p_idx] (including itself)."""
    return [i for i, q in enumerate(points) if euclidean(points[p_idx], q) <= eps]


def dbscan(points, eps, min_pts):
    """
    Parameters
    ----------
    points   : list of (x, y) tuples
    eps      : neighbourhood radius ε
    min_pts  : minimum points (including self) to be a core point

    Returns
    -------
    labels : list[int]   –1 = noise, 0, 1, … = cluster id
    types  : list[str]   'core' | 'border' | 'noise'
    """
    n       = len(points)
    labels  = [-1] * n
    types   = ['noise'] * n
    visited = [False] * n
    cluster_id = 0

    for i in range(n):
        if visited[i]:
            continue

        visited[i] = True
        neighbors = region_query(points, i, eps)

        # Not enough neighbours → tentatively noise (may be upgraded to border later)
        if len(neighbors) < min_pts:
            continue                         # label stays -1

        # ── Core point: seed a new cluster ──────────────────────────────────
        types[i]  = 'core'
        labels[i] = cluster_id

        # FIX 1: use a deque + a set so "already queued" check is O(1)
        queue  = deque(neighbors)          # frontier to expand
        queued = set(neighbors)            # O(1) membership test

        while queue:
            q_idx = queue.popleft()        # FIX 2: popleft, not index-walk

            # FIX 3: assign cluster label regardless of visited status
            #        (a point can be in the queue but not yet visited)
            if labels[q_idx] == -1:        # unvisited or was tentative noise
                labels[q_idx] = cluster_id

            if visited[q_idx]:
                continue

            visited[q_idx] = True
            q_neighbors = region_query(points, q_idx, eps)

            if len(q_neighbors) >= min_pts:
                # q is also a core point → expand frontier
                types[q_idx] = 'core'
                for nb in q_neighbors:
                    if nb not in queued:   # O(1) set check
                        queued.add(nb)
                        queue.append(nb)
            else:
                # q has too few neighbours → border of this cluster
                types[q_idx] = 'border'

        cluster_id += 1

    # Any point still labelled -1 is genuine noise
    for i in range(n):
        if labels[i] == -1:
            types[i] = 'noise'

    return labels, types

sklearn_dbscan.py 
#!/usr/bin/env python3
"""Production DBSCAN using scikit-learn."""

import numpy as np
from sklearn.cluster import DBSCAN


# ── 1. Dataset ────────────────────────────────────────────────────────────
POINTS = np.array([
    [220,230], [250,210], [250,250], [220,270], [190,250],
    [190,210], [290,190], [290,270], [220,310], [150,270],
    [150,190], [220,180], [560,230], [590,210], [590,250],
    [560,270], [530,250], [530,210], [630,190], [630,270],
    [560,310], [490,270], [490,190], [560,180], [390,100],
    [390,360], [80, 150], [80, 310], [700,150], [700,310]
])


# ── 2. Fit ────────────────────────────────────────────────────────────────
# eps=45 matches the ε-circle in the visualization.
# min_samples=4 counts the point itself + 3 neighbours.
db = DBSCAN(eps=45, min_samples=4).fit(POINTS)

labels = db.labels_   # -1 = noise


# ── 3. Results ───────────────────────────────────────────────────────────
n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
n_noise    = (labels == -1).sum()

print(f"Clusters : {n_clusters}")   # → 2
print(f"Noise    : {n_noise}")       # → 6
print("Labels   :", labels)


# ── 4. Core point mask ───────────────────────────────────────────────────
core_mask = np.zeros(len(POINTS), dtype=bool)
core_mask[db.core_sample_indices_] = True

print("Core pts :", core_mask.astype(int))
